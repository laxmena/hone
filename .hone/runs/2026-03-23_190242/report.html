import sys
import heapq
from collections import defaultdict

class HyperLogLog:
    """Probabilistic cardinality estimator with minimal memory overhead."""
    def __init__(self, p=8):
        self.p = p
        self.m = 1 << p  # 2^p registers
        self.registers = [0] * self.m
        self.alpha = 0.7213 / (1 + 1.079 / self.m)
    
    def add(self, value):
        """Add an element using hash-based probabilistic counting."""
        h = hash(value) & 0xffffffff
        j = h >> (32 - self.p)
        w = h & ((1 << (32 - self.p)) - 1)
        left_most_one = (w & -w).bit_length() if w else 33 - self.p
        self.registers[j] = max(self.registers[j], left_most_one)
    
    def cardinality(self):
        """Estimate cardinality from registers."""
        raw = self.alpha * self.m * self.m / sum(2.0 ** (-x) for x in self.registers)
        return int(raw)

def analyze_logs(file_path):
    """
    Reads a web server log file and returns the top 3 most visited endpoints
    and the number of unique IP addresses that visited them.
    """
    endpoint_hll = defaultdict(HyperLogLog)
    
    with open(file_path, 'r') as f:
        for line in f:
            # Quick validation: need at least IP + request line
            if '"' not in line:
                continue
            
            # Extract IP (first space-delimited token)
            space_idx = line.find(' ')
            if space_idx <= 0:
                continue
            ip = line[:space_idx]
            
            # Extract endpoint from quoted request
            try:
                q1 = line.index('"')
                q2 = line.index('"', q1 + 1)
                request = line[q1 + 1:q2]
                parts = request.split()
                if len(parts) >= 2:
                    endpoint = parts[1]
                    endpoint_hll[endpoint].add(ip)
            except (ValueError, IndexError):
                pass
    
    # Return top 3 endpoints with estimated cardinality
    return heapq.nlargest(
        3,
        ((ep, hll.cardinality()) for ep, hll in endpoint_hll.items()),
        key=lambda x: x[1]
    )

if __name__ == "__main__":
    if len(sys.argv) > 1:
        print(analyze_logs(sys.argv[1]))

LOG_PATTERN = re.compile(r'"[A-Z]+\s+([^\s]+)\s+')
